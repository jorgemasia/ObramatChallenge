package com.obramat.pedidos.application.usecases;

import com.obramat.pedidos.domain.model.OrderDetail;
import com.obramat.pedidos.domain.ports.OrderDetailRepository;
import java.util.List;
import java.util.stream.Collectors;

public class ListDetailsByOrderUseCase {
    private final OrderDetailRepository repo;
    public ListDetailsByOrderUseCase(OrderDetailRepository repo) { this.repo = repo; }

    public List<OrderDetail> execute(Long orderId) {
        // use Stream to potentially transform/filter
        return repo.findByOrderId(orderId)
                .stream()
                .collect(Collectors.toList());
    }
}
